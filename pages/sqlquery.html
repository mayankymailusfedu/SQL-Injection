SQL Query

Let's start with refreshing our sql query knowledge

SHOW DATABASES - list all the existing databases in the server

CREATE DATABASE DBNAME - creates a new database

DROP DATABASE DBNAME - deletes the specified database

CREATE DATABASE IF NOT EXISTS DBNAME - creates a new database, if it does not exist

CREATE TABLE TABLENAME (ColumnName Datatype, ...) - creates a new table

DROP TABLE TABLENAME - deletes the specified table

DESCRIBE TABLENAME - describes fields of the table

INSERT INTO TABLENAME VALUES ('AnyValue', ...) - insert records into table

SELECT * FROM TABLENAME - display records of the table

DELETE FROM TABLENAME WHERE ColumnName = ColumnValue - delete a specific record

SELECT ColumnName1, ColumnName2,... FROM TABLENAME - display specific column of the table

SELECT ColumnName1, ColumnName2,... FROM TABLENAME WHERE CRITERIA - display specific column with specific record/records of the table

UPDATE TABLENAME SET ColumnName = NewValue;

SELECT ColumnName FROM TABLENAME1 UNION SELECT Column_Name FROM TABLENAME2 - union operator is used to combine the result-set of two or more SELECT statements
